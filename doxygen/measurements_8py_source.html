49  def __init__(self, name, value, event=None, override_unit=False, **kwargs):
50  """Construct a new Measurement with the given name and value.
51 
52  Args:
53  name (str): The Measurement's generic name in OpenXC.
54  value (str, float, or bool): The Measurement's value.
55 
56  Kwargs:
57  event (str, bool): An optional event for compound Measurements.
58  override_unit (bool): The value will be coerced to the correct units
59  if it is a plain number.
60 
61  Raises:
62  UnrecognizedMeasurementError if the value is not the correct units,
63  e.g. if it's a string and we're expecting a numerical value
64 
65  @author Christopher Peplin github@rhubarbtech.com
66  @date June 25, 2013
67  @version 0.9.4
68 
69  @param name The name of the measurement
70  @param value The value of the measurement
71  @param event (str, bool): An optional event for compound Measurements.
72  @param override_unit The override unit
73  @param kwargs The kwargs instance.
74  @exception UnrecognizedMeasurementError if the value is not the correct
75  units, e.g. if it's a string and we're expecting a numerical value.
76  """
77  super(Measurement, self).__init__()
78  self.name = name
79  if self.unit != units.Undefined and override_unit:
80  if type(value) == unicode:
81  raise UnrecognizedMeasurementError("%s value cannot be a string"
82  % self.__class__)
83  value = self.unit(value)
84  self.value = value
85  self.event = event

111  @classmethod
112  def from_dict(cls, data):
113  """Create a new Measurement subclass instance using the given dict.
114 
115  If Measurement.name_from_class was previously called with this data's
116  associated Measurement sub-class in Python, the returned object will be
117  an instance of that sub-class. If the measurement name in ``data`` is
118  unrecognized, the returned object will be of the generic ``Measurement``
119  type.
120 
121  Args:
122  data (dict): the data for the new measurement, including at least a
123  name and value.
124 
125  @param cls the object instance.
126  @param data the data object instance.
127  """
128  measurement_class = cls._class_from_name(data['name'])
129  args = [data['value']]
130  if measurement_class == Measurement:
131  args.insert(0, data['name'])
132  return measurement_class(*args, event=data.get('event', None),
133  override_unit=True)

135  @classmethod
136  def name_from_class(cls, measurement_class):
137  """For a given measurement class, return its generic name.
138 
139  The given class is expected to have a ``name`` attribute, otherwise this
140  function will raise an execption. The point of using this method instead
141  of just trying to grab that attribute in the application is to cache
142  measurement name to class mappings for future use.
143 
144  Returns:
145  the generic OpenXC name for a measurement class.
146 
147  Raise:
148  UnrecognizedMeasurementError: if the class does not have a valid
149  generic name
150 
151  @param cls the object instance.
152  @param measurement_class the measurement class object instance.
153  """
154  try:
155  name = getattr(measurement_class, 'name')
156  except AttributeError:
157  raise UnrecognizedMeasurementError()
158  else:
159  cls._measurement_map[name] = measurement_class
160  return name
