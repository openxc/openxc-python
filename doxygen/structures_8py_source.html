378  def translate(self, can_data):
379  """Translate Routine"""
380  translated_value = int(can_data, 16)
381 
382  # Trim off everything to the right
383  translated_value >>= 64 - self.bit_position - self.bit_size
384  mask = int(math.pow(2, self.bit_size)) - 1
385  translated_value &= mask
386  translated_value *= self.factor
387  translated_value += self.offset
388  return translated_value
