15 class DataSource(threading.Thread):
16  """Interface for all vehicle data sources. This inherits from Thread and
17  when a source is added to a vehicle it attempts to call the ``start()``
18  method if it exists. If an implementer of DataSource needs some background
19  process to read data, it's just a matter of defining a ``run()`` method.
20 
21  A data source requires a callback method to be specified. Whenever new data
22  is received, it will pass it to that callback.

82  def run(self):
83  """Continuously read data from the source and attempt to parse a valid
84  message from the buffer of bytes. When a message is parsed, passes it
85  off to the callback if one is set.
86  """
87  message_buffer = b""
88  while True:
89  message_buffer += self._read()
90  while True:
91  message, message_buffer, byte_count = self._parse_message(
92  message_buffer)
93  if message is None:
94  break
95  if not hasattr(message, '__iter__') or not (
96  ('name' in message and 'value' in message) or (
97  'id' in message and 'data' in message)):
98  self.corrupted_messages += 1
99  break
100 
101  self.bytes_received += byte_count
102  if self.callback is not None:
103  self.callback(message,
104  data_remaining=len(message_buffer) > 0)

106  def _parse_message(self, message_buffer):
107  """If a message can be parsed from the given buffer, return it and
108  remove it.
109 
110  Returns the message if one could be parsed, otherwise None, and the
111  remainder of the buffer.
112 
113  @param message_buffer The message_buffer object instance.
114  """
115  if not isinstance(message_buffer, bytes):
116  message_buffer = message_buffer.encode("utf-8")
117  parsed_message = None
118  remainder = message_buffer
119  message = ""
120  if b"\n" in message_buffer:
121  message, _, remainder = message_buffer.partition(b"\n")
122  try:
123  parsed_message = JsonFormatter.deserialize(message)
124  if not isinstance(parsed_message, dict):
125  raise ValueError()
126  except ValueError:
127  pass
128  return parsed_message, remainder, len(message)
