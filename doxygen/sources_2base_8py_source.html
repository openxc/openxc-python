9  when a source is added to a vehicle it attempts to call the ``start()``
10  method if it exists. If an implementer of DataSource needs some background
11  process to read data, it's just a matter of defining a ``run()`` method.
12 
13  A data source requires a callback method to be specified. Whenever new data
14  is received, it will pass it to that callback.
15  """

68  def run(self):
69  """Continuously read data from the source and attempt to parse a valid
70  message from the buffer of bytes. When a message is parsed, passes it
71  off to the callback if one is set.
72  """
73  message_buffer = b""
74  while True:
75  message_buffer += self._read()
76  while True:
77  message, message_buffer, byte_count = self._parse_message(
78  message_buffer)
79  if message is None:
80  break
81  if not hasattr(message, '__iter__') or not (
82  ('name' in message and 'value' in message) or (
83  'id' in message and 'data' in message)):
84  self.corrupted_messages += 1
85  break
86 
87  self.bytes_received += byte_count
88  if self.callback is not None:
89  self.callback(message,
90  data_remaining=len(message_buffer) > 0)

92  def _parse_message(self, message_buffer):
93  """If a message can be parsed from the given buffer, return it and
94  remove it.
95 
96  Returns the message if one could be parsed, otherwise None, and the
97  remainder of the buffer.
98 
99  @param message_buffer The message_buffer object instance.
100  """
101  if not isinstance(message_buffer, bytes):
102  message_buffer = message_buffer.encode("utf-8")
103  parsed_message = None
104  remainder = message_buffer
105  message = ""
106  if b"\n" in message_buffer:
107  message, _, remainder = message_buffer.partition(b"\n")
108  try:
109  parsed_message = JsonFormatter.deserialize(message)
110  if not isinstance(parsed_message, dict):
111  raise ValueError()
112  except ValueError:
113  pass
114  return parsed_message, remainder, len(message)
