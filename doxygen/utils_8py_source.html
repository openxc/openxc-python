82 def merge(a, b):
83  """Merge two deep dicts non-destructively
84 
85  Uses a stack to avoid maximum recursion depth exceptions
86 
87  >>> a = {'a': 1, 'b': {1: 1, 2: 2}, 'd': 6}
88  >>> b = {'c': 3, 'b': {2: 7}, 'd': {'z': [1, 2, 3]}}
89  >>> c = merge(a, b)
90  >>> from pprint import pprint; pprint(c)
91  {'a': 1, 'b': {1: 1, 2: 7}, 'c': 3, 'd': {'z': [1, 2, 3]}}
92  @param a The a input to merge with the b input.
93  @param b The b input to merge with the a input.
94  @return list List object containing the merge a and b inputs.
95  """
96  assert quacks_like_dict(a), quacks_like_dict(b)
97  dst = a.copy()
98 
99  stack = [(dst, b)]
100  while stack:
101  current_dst, current_src = stack.pop()
102  for key in current_src:
103  if key not in current_dst:
104  current_dst[key] = current_src[key]
105  else:
106  if (quacks_like_dict(current_src[key]) and
107  quacks_like_dict(current_dst[key])):
108  stack.append((current_dst[key], current_src[key]))
109  elif (quacks_like_list(current_src[key]) and
110  quacks_like_list(current_dst[key])):
111  current_dst[key].extend(current_src[key])
112  else:
113  current_dst[key] = current_src[key]
114  return dst
