60 def total_seconds(delta):
61  """ timedelta.total_seconds() is only in 2.7, so we backport it here for
62  2.6
63  @param delta The time delta.
64  """
65  return (delta.microseconds + (delta.seconds

223  def _redraw(self):
224  """The redraw routine"""
225  self.window.erase()
226  max_rows = self.window.getmaxyx()[0] - 4
227  for row, element in enumerate(sorted(self.elements.values(),
228  key=lambda elt: elt.measurement_type.name)):
229  if row > max_rows:
230  break
231  element.print_to_window(self.window, row, self.started_time)
232  self.window.addstr(max_rows + 1, 0,
233  "Message count: %d (%d corrupted)" % (self.messages_received,
234  self.source.corrupted_messages), curses.A_REVERSE)
235  self.window.addstr(max_rows + 2, 0,
236  "Total received: %s" %
237  sizeof_fmt(self.source.bytes_received),
238  curses.A_REVERSE)
239  self.window.addstr(max_rows + 3, 0, "Data Rate: %s" %
240  sizeof_fmt(self.source.bytes_received /
241  (total_seconds(datetime.now() - self.started_time)
242  + 0.1)),
243  curses.A_REVERSE)
244  self.window.refresh()
